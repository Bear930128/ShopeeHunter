async function toggleCheck(docId){ const item = items.find(i => i.docId === docId); const newValue = !item.checked; await fetch( `https://firestore.googleapis.com/v1/projects/${PROJECT_ID}/databases/(default)/documents/${COLLECTION}/${docId}?updateMask.fieldPaths=checked`, { method:"PATCH", headers:{"Content-Type":"application/json"}, body: JSON.stringify({ fields:{ checked:{ booleanValue:newValue } } }) } ); // ⭐ 這行很重要（強制刷新） await loadData(); // ⭐ 保險再render一次 render(items); }
