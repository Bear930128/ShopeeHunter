async function loadData(){ const res = await fetch(`https://firestore.googleapis.com/v1/projects/${PROJECT_ID}/databases/(default)/documents/${COLLECTION}`); const data = await res.json(); if(!data.documents){ items = []; renderDashboard(); return; } const raw = data.documents.map(doc=>{ const f = doc.fields || {}; const sold = f.sold?.stringValue || "0"; const fans = f.fans?.stringValue || "0"; return { docId: doc.name.split("/").pop(), title: f.title?.stringValue || "", shop: f.shop?.stringValue || "", sold, fans, company: f.company?.stringValue || "", taxId: f.taxId?.stringValue || "", url: f.url?.stringValue || "", checked: f.checked?.booleanValue || false, time: f.time?.stringValue || "", score: parseNum(sold)*0.7 + parseNum(fans)*0.3 }; }); // ⭐⭐⭐ 強制去重（最終穩定版）⭐⭐⭐ const map = new Map(); raw.forEach(item => { // ❗ 沒 URL 直接跳過（避免壞資料） if(!item.url) return; const key = item.url.trim(); if(!map.has(key)){ map.set(key, item); }else{ const old = map.get(key); // ⭐ 優先保留已標記 if(item.checked && !old.checked){ map.set(key, item); return; } // ⭐ 再用時間決定（最新） if((item.time || "") > (old.time || "")){ map.set(key, item); } } }); items = Array.from(map.values()); // ⭐ 排序（避免畫面亂） items.sort((a,b)=>b.score - a.score); renderDashboard(); // ⭐ 如果在列表頁 → 重新渲染 if(currentPage !== "dashboard"){ applySearch(); } }
