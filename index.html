function getPageType() { const url = window.location.href; // 商品頁：含有 i.數字.數字（蝦皮商品ID格式） if (url.match(/i\.\d+\.\d+/)) return 'product'; // 商品頁：路徑層數多（店名/商品名） const path = url.replace(/https?:\/\/[^/]+/, '').split('?')[0]; const parts = path.split('/').filter(Boolean); if (parts.length >= 2) return 'product'; // 賣場頁：含 /shop/ 或路徑只有一段 if (url.includes('/shop/') || parts.length <= 1) return 'shop'; return 'unknown'; } function parseSalesNum(str) { if (!str) return 0; const s = str.replace(/,/g, ''); if (/萬/.test(s)) return Math.round(parseFloat(s) * 10000); if (/k/i.test(s)) return Math.round(parseFloat(s) * 1000); return parseInt(s) || 0; } function scrapeProduct() { const result = { type: 'product', url: window.location.href, product: '', sales: 0 }; const nameEls = document.querySelectorAll('h1'); for (const el of nameEls) { const t = el.textContent.trim(); if (t.length > 2) { result.product = t; break; } } const allEls = Array.from(document.querySelectorAll('*')); for (const el of allEls) { if (el.children.length > 0) continue; const t = el.textContent.trim(); const m = t.match(/^([\d,\.]+[萬k]?)\s*(已售出|sold)/i); if (m) { result.sales = parseSalesNum(m[1]); break; } } return result; } function scrapeShop() { const result = { type: 'shop', shopName: '', fans: 0, company: '無', taxId: '無' }; const h1 = document.querySelector('h1'); if (h1) result.shopName = h1.textContent.trim(); const bodyText = document.body.innerText; const allEls = Array.from(document.querySelectorAll('*')); for (const el of allEls) { if (el.children.length > 0) continue; const t = el.textContent.trim(); if (/粉絲|follower/i.test(t)) { const parent = el.closest('[class]') || el.parentElement; if (parent) { const nums = parent.textContent.match(/[\d,\.]+[萬k]?/g); if (nums) { result.fans = parseSalesNum(nums[0]); break; } } } } const taxM = bodyText.match(/統[一]?編[號]?[：:]\s*(\d{8})|(\d{8})/); if (taxM) result.taxId = taxM[1] || taxM[2]; const compM = bodyText.match(/([^\n\r]{2,20}(?:股份)?有限公司|[^\n\r]{2,10}企業社|[^\n\r]{2,10}工作室)/); if (compM && compM[1].length <= 30) result.company = compM[1].trim(); return result; } chrome.runtime.onMessage.addListener((msg, sender, sendResponse) => { if (msg.action === 'getPageType') sendResponse({ pageType: getPageType() }); if (msg.action === 'scrapeProduct') sendResponse(scrapeProduct()); if (msg.action === 'scrapeShop') sendResponse(scrapeShop()); return true; });
