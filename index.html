async function loadData(){ try{ const res = await fetch(`https://firestore.googleapis.com/v1/projects/${PROJECT_ID}/databases/(default)/documents/${COLLECTION}`); const data = await res.json(); // ⭐ 防呆（這行很重要） if(!data || !data.documents){ items = []; renderDashboard(); return; } const raw = data.documents.map(doc=>{ const f = doc.fields || {}; const sold = f.sold?.stringValue || "0"; const fans = f.fans?.stringValue || "0"; return { docId: doc.name.split("/").pop(), title: f.title?.stringValue || "", shop: f.shop?.stringValue || "", sold, fans, company: f.company?.stringValue || "", taxId: f.taxId?.stringValue || "", url: f.url?.stringValue || "", checked: f.checked?.booleanValue || false, time: f.time?.stringValue || "", score: parseNum(sold)*0.7 + parseNum(fans)*0.3 }; }); // ⭐⭐⭐ 去重（最穩版本）⭐⭐⭐ const map = new Map(); raw.forEach(item => { if(!item.url) return; const key = item.url.trim(); if(!map.has(key)){ map.set(key, item); }else{ const old = map.get(key); // ⭐ 優先保留已標記 if(item.checked && !old.checked){ map.set(key, item); return; } // ⭐ 用時間判斷（一定要） if((item.time || "") > (old.time || "")){ map.set(key, item); } } }); items = Array.from(map.values()); items.sort((a,b)=>b.score - a.score); renderDashboard(); if(currentPage !== "dashboard"){ applySearch(); } }catch(e){ console.log("❌ loadData錯誤", e); items = []; renderDashboard(); } }
